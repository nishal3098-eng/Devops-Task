# Custom Nginx Image
FROM nginx:alpine

LABEL maintainer="nishal3098-eng"
LABEL description="Custom Nginx image for Docker Task 3"

# Remove default nginx welcome page
RUN rm -rf /usr/share/nginx/html/*

# Create the bind mount target directory
RUN mkdir -p /var/opt/nginx

# Custom nginx config to serve from /var/opt/nginx
RUN echo 'server { \
    listen 80; \
    server_name _; \
    root /var/opt/nginx; \
    index index.html; \
    location / { \
        try_files $uri $uri/ =404; \
    } \
}' > /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
